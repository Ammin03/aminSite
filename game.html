import random # Variables globales pregunta_actual = "" respuesta_correcta = 0 def generar_pregunta(): global pregunta_actual, respuesta_correcta operadores = ['+', '-', '*', '/'] operador = random.choice(operadores) num1 = random.randint(1, 10) num2 = random.randint(1, 10) # Asegurarse de que no haya división por cero if operador == '/': num1 = num1 * num2 # Garantiza que sea una división exacta pregunta_actual = f"{num1} {operador} {num2}" respuesta_correcta = round(eval(pregunta_actual), 2) # Muestra la pregunta en la página pregunta_element = Element("pregunta") pregunta_element.element.innerText = f"¿Cuánto es {pregunta_actual}?" def verificar_respuesta(*args): # Obtiene la respuesta del usuario respuesta_usuario = float(Element("respuesta").element.value) resultado_element = Element("resultado") # Verifica si es correcta o incorrecta if respuesta_usuario == respuesta_correcta: resultado_element.element.innerText = "¡Correcto!" resultado_element.element.style.color = "green" else: resultado_element.element.innerText = f"Incorrecto. La respuesta era {respuesta_correcta}." resultado_element.element.style.color = "red" # Conecta los botones a las funciones Element("nuevaPregunta").element.onclick = generar_pregunta Element("verificar").element.onclick = verificar_respuesta
